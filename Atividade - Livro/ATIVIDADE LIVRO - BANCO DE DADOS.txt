/* Lógico_Emprestimo_Livros: */

-- CRIAÇÃO TABELAS
CREATE TABLE Autor (
    id_autor serial PRIMARY KEY,
    nome_autor varchar(50) not null
);

CREATE TABLE Livro (
    id_livro serial PRIMARY KEY,
    titulo varchar(30) not null,
    fk_autor_id inteiro,
	foreign key (fk_autor_id) references Autor (id_autor)
    on delete cascade on update no action
);


CREATE TABLE Usuario (
    id_usuario serial PRIMARY KEY,
    nome_usuario varchar(50) not null,
    email varchar(30) unique 
);

CREATE TABLE Livro_Usuario_Emprestimo (
    id serial PRIMARY KEY,
    data_emprestimo date,
    data_devolucao date,
    valor numeric,
    fk_livro_id inteiro,
    fk_usuario_id inteiro,
    foreign key (fk_livro_id) references Livro (id_livro)
    on delete cascade on update no action,
    foreign key (fk_usuario_id) references Usuario (id_usuario)
    on delete cascade on update no action
);

/*QUESTÃO 01*/
select e.data_emprestimo, l.titulo, u.nome_usuario from Livro_Usuario_Emprestimo e
inner join usuario u 
on u.id_usuario = e.fk_usuario_id
inner join livro l
on e.fk_livro_id = l.id_livro
where data_devolucao isnull;

/*QUESTÃO 02*/
select l.titulo, a.nome_autor  from livro l 
left join Livro_Usuario_Emprestimo e 
on l.id_livro = e.fk_livro_id
inner join autor a
on l.fk_autor_id  = a.id_autor 	
where e.id isnull;

/*QUESTÃO 03*/
select count(l.id_livro) as Livros_emprestados from Livro_Usuario_Emprestimo e
inner join usuario u 
on u.id_usuario = e.fk_usuario_id
inner join livro l
on e.fk_livro_id = l.id_livro

/*QUESTÃO 04*/

inner join autor a
on l.fk_autor_id = a.id_autor
group by a.id_autor
